217double medium(student &tmp) {
218 vector<int> grd = tmp.getGrade(); // temporary copy of grades
219 int sum = accumulate(grd.begin(), grd.end(), sum); // computing the sum of the given initial value and the elements in the given range

222double median(student &tmp) {
223 vector<int> grd = tmp.getGrade(); // temporary copy of grades
224 sort(grd.begin(), grd.end());
